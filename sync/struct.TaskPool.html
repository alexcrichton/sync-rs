A task pool used to execute functions in parallel.

Spawns n worker tasks and replenishes the pool if any worker tasks panic.

Example


let pool = TaskPool::new(4u);

let (tx, rx) = channel();
for _ in range(0, 8u) {
    let tx = tx.clone();
    pool.execute(proc() {
        tx.send(1u);
    });
}

assert_eq!(rx.iter().take(8u).sum(), 8u);
